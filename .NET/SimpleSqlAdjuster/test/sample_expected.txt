FetchUsers
SELECT
  USER_ID,
  USER_NAME
FROM
  USERS
WHERE
  USER_TYPE = :type
  AND
  STATUS = 'ACTIVE'

SELECT
  *
FROM
  EMPLOYEES E
WHERE
  E.DEPT_ID = :deptId

Orders
SELECT
  ORDER_ID
FROM
  ORDERS
:_WHERE_AND
(
  ORDER_DATE >= :fromDate,
  ORDER_DATE <= :toDate,
  GCOM = :GCOM
)

SubQueryExample
SELECT
  C.CUSTOMER_ID,
  (
    SELECT
      COUNT(*)
    FROM
      ORDERS O
    WHERE
      O.CUSTOMER_ID = C.CUSTOMER_ID
      AND
      O.STATUS IN('NEW', 'HOLD')
  ) AS ORDER_COUNT
FROM
  CUSTOMERS C
WHERE
  EXISTS
  (
    SELECT
      1
    FROM
      PAYMENTS P
    WHERE
      P.CUSTOMER_ID = C.CUSTOMER_ID
      AND
      P.AMOUNT > 1000
  )

UpdateSimple
UPDATE
  EMPLOYEES
SET
  STATUS = 'ACTIVE',
  UPDATED_AT = SYSDATE
WHERE
  EMP_ID = :empId

UpdateWithSubquery
UPDATE
  EMPLOYEES
SET
  STATUS = 'INACTIVE'
WHERE
  EMP_ID IN
  (
    SELECT
      EMP_ID
    FROM
      TERMINATION_REQUESTS
    WHERE
      REQUEST_DATE > SYSDATE - 30
  )

InsertSimple
INSERT INTO
  SALES
(
  SALE_ID,
  AMOUNT,
  CREATED_AT
)
VALUES
(
  :saleId,
  :amount,
  SYSDATE
)

InsertWithSubquery
INSERT INTO
  SALES_SUMMARY
(
  CUSTOMER_ID,
  TOTAL_AMOUNT
)
SELECT
  CUSTOMER_ID,
  SUM(AMOUNT)
FROM
  SALES_DETAIL
WHERE
  SALE_DATE BETWEEN :from
  AND
  :to
GROUP BY
  CUSTOMER_ID

DeleteSimple
DELETE
FROM
  SESSIONS
WHERE
  EXPIRES_AT < SYSDATE

DeleteWithSubquery
DELETE
FROM
  SESSIONS
WHERE
  USER_ID IN
  (
    SELECT
      USER_ID
    FROM
      USERS
    WHERE
      STATUS = 'LOCKED'
  )

FromSubquery
SELECT
  T.ORDER_ID,
  T.AMOUNT
FROM
  (
    SELECT
      ORDER_ID,
      AMOUNT
    FROM
      ORDERS
    WHERE
      STATUS = 'PAID'
  ) T
WHERE
  T.AMOUNT > 1000
