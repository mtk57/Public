Orders

  ORDER_ID	ORDER_ID
FROM
  ORDERS	ORDERS
:_WHERE_AND
(
  ORDER_DATE >= :fromDate,	ORDER_DATE
  ORDER_DATE <= :toDate,	ORDER_DATE

SELECT
  C.CUSTOMER_ID,	CUSTOMER_ID
  (
    SELECT
      COUNT(*)	
    FROM
      ORDERS O	ORDERS
    WHERE
      O.CUSTOMER_ID = C.CUSTOMER_ID	CUSTOMER_ID
      AND
      O.STATUS IN('NEW', 'HOLD')	STATUS
  ) AS ORDER_COUNT
FROM
  CUSTOMERS C	CUSTOMERS
WHERE
  EXISTS
  (
    SELECT
      1	
    FROM
      PAYMENTS P	PAYMENTS
    WHERE
      P.CUSTOMER_ID = C.CUSTOMER_ID	CUSTOMER_ID
      AND
      P.AMOUNT > 1000	AMOUNT
  )
